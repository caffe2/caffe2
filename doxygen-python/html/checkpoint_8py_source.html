33 @context.define_context()
34 class Job(object):
35  """
36  A Job defines three TaskGroups: the `init_group`, the `epoch_group` and the
37  `exit_group` which will be run by a JobRunner.
38 
39  The `init_group` will be run only once at startup. Its role is to
40  initialize globally persistent blobs such as model weights, accumulators
41  and data file lists.
42 
43  The `epoch_group` will be run in a loop after init_group. The loop will
44  exit when any of the stop signals added with `add_stop_signal` is True
45  at the end of an epoch.
46 
47  The `exit_group` will be run only once at the very end of the job, when one
48  of the stopping criterias for `epoch_group` was met. The role of this group
49  is save the results of training in the end of the job.
50 
51  Jobs are context-driven, so that Tasks can be added to the active Job
52  without having to explicitly pass the job object around.
53 
54  Example of usage:
55 
56  def build_reader(partitions):
57  with Job.current().init_group:
58  reader = HiveReader(init_reader, ..., partitions)
59  Task(step=init_reader)
60  with Job.current().epoch_group:
61  limited_reader = ReaderWithLimit(reader, num_iter=10000)
62  data_queue = pipe(limited_reader, num_threads=8)
63  Job.current().add_stop_signal(limited_reader.data_finished())
64  return data_queue
65 
66  def build_hogwild_trainer(reader, model):
67  with Job.current().init_group:
68  Task(step=model.param_init_net)
69  with Job.current().epoch_group:
70  pipe(reader, processor=model, num_threads=8)
71  with Job.current().exit_group:
72  Task(step=model.save_model_net)
73 
74  with Job() as job:
75  reader = build_reader(partitions)
76  model = build_model(params)
77  build_hogwild_trainer(reader, model)
78  """
79  def __init__(self,
80  init_group=None, epoch_group=None,
81  exit_group=None, stop_signals=None,
82  nodes_to_checkpoint=None):
83  self.init_group = init_group or TaskGroup(
84  workspace_type=WorkspaceType.GLOBAL)
85  self.epoch_group = epoch_group or TaskGroup()
86  self.exit_group = exit_group or TaskGroup()
87  self.stop_signals = stop_signals or []
88  self._nodes_to_checkpoint = nodes_to_checkpoint
89 
90  def nodes_to_checkpoint(self):
91  if self._nodes_to_checkpoint:
92  return self._nodes_to_checkpoint
93  else:
94  return self.init_group.used_nodes()

97  return Job(
98  init_group=session_class.compile(self.init_group),
99  epoch_group=session_class.compile(self.epoch_group),
100  exit_group=session_class.compile(self.exit_group),
101  stop_signals=self.stop_signals,
102  nodes_to_checkpoint=self.nodes_to_checkpoint())

153 class CheckpointManager(object):
154  """
155  Controls saving and loading of workspaces on every epoch boundary of a job.
156  If a CheckpointManager instance is passed to JobRunner, then JobRunner will
157  call `init`, `read` and `save` at different moments in between epoch runs.
158 
159  Args:
160  db_prefix: The prefix used to construct full db name. Since `absolute_path`
161  is set to True, this will be used as db_name in SaveOp.
162  node_name: Name of the node where this checkpoint_manager is used.
163  db_type: Type of database to use for storing checkpoint.
164  metadata_handler: An optional object capable of reading/writing
165  checkpoint info in storage of choice.
166  """
167  def __init__(self, db_prefix, node_name, db_type, metadata_handler=None):
168  self._db_prefix = db_prefix
169  self._node_name = node_name
170  self._db_type = db_type
171  self._metadata_handler = metadata_handler
172  # make sure these blobs are the first in the checkpoint file.
173  self._net = core.Net('!!checkpoint_mngr')
174  self._blob_names = self._net.AddExternalInput('blob_names')
175  self._names_output = None
176 
177  """
178  Initialize the checkpoint manager. Determines all blobs that need to be saved
179  or loads from a checkpoint.
180 
181  Args:
182  nodes: An array of nodes where this checkpoint manager is running. Should
183  only contain a single node.
184  retrieve_from_epoch: Set to a number to load blobs from this epoch.
185  path_prefix: Used to construct db name or path where checkpoint files are
186  stored.
187  path_type: Indicate the type of path where checkpoint files are stored.
188  """
189  def init(
190  self,
191  nodes=None,
192  retrieve_from_epoch=None,
193  path_prefix=None,
194  path_type=None
195  ):
196  """
197  Build a Task that will be run once after the job's `init_group` is run.
198  This task will determine which blobs need to be checkpointed.
199  If retrieve_from_epoch is not None, then the checkpoint metadata is
200  retrieved from a previously saved checkpoint.
201  """
202  assert nodes is None or len(nodes) == 1, (
203  'CheckpointManager only supports single node.')
204 
205  self._path_prefix = path_prefix
206  self._path_type = path_type
207 
208  with Task(outputs=[self._blob_names]) as task:
209  if retrieve_from_epoch is None:
210  ops.GetAllBlobNames(
211  [],
212  self._blob_names,
213  include_shared=False)
214  else:
215  full_db_name = db_name(retrieve_from_epoch,
216  self._node_name, self._db_prefix, path_prefix)
217  db_type = path_type or self._db_type
218  logger.info("Initializing checkpoints from = %s"
219  % full_db_name)
220  ops.Load(
221  [], self._blob_names,
222  db=full_db_name,
223  db_type=db_type,
224  absolute_path=True)
225  self._names_output = task.outputs()[0]
226  return task
227 
228  def blob_list(self):
229  assert self._names_output
230  return self._names_output.fetch().tolist()

232  def load(self, epoch, path_prefix=None, path_type=None):
233  """
234  Build a Task that will be run by JobRunner when the job is to be
235  resumed from a given epoch. This task will run a Load op that will
236  load and deserialize all relevant blobs from a persistent storage.
237  """
238  full_db_name = db_name(epoch, self._node_name, self._db_prefix, path_prefix)
239  db_type = path_type or self._db_type
240  logger.info("Loading checkpoints from = %s" % full_db_name)
241  with Task() as task:
242  ops.Load(
243  [],
244  self.blob_list(),
245  db=full_db_name,
246  db_type=db_type,
247  absolute_path=True)
248  return task

250  def load_blobs_from_checkpoint(self, blob_names, epoch):
251  """
252  Builds a Task that loads only the necessary blobs from a checkpoint of
253  the given epoch. The necessary blobs are given in the blob_names
254  argument.
255 
256  Args:
257  blob_names: A list of strings. Each string is the name of a
258  blob.
259  epoch: The checkpoint epoch to load from.
260 
261  Returns:
262  A Task which loads the specified blobs from the checkpoint of the
263  given epoch.
264  """
265  logger.info('Load from %s' % db_name(epoch, self._node_name, self._db_prefix))
266  with Task() as task:
267  ops.Load(
268  [],
269  blob_names,
270  db=db_name(epoch, self._node_name, self._db_prefix),
271  db_type=self._db_type,
272  absolute_path=True,
273  allow_incomplete=True)
274  return task

322 class MultiNodeCheckpointManager(object):
323  """
324  Coordinates checkpointing and checkpointing across multiple nodes.
325  Each of `init`, `load` and `save` will build TaskGroups which will
326  trigger checkpointing on each of the nodes involved in a distributed job.
327 
328  Args:
329  db_prefix: The prefix used to construct full db name. Since `absolute_path`
330  is set to True, this will be used as db_name in SaveOp.
331  db_type: Type of database to use for storing checkpoint.
332  metadata_handler: An optional object capable of reading/writing
333  checkpoint info in storage of choice.
334  """
335  def __init__(self, db_prefix, db_type, metadata_handler=None):
336  self._node_managers = None
337  self._db_prefix = db_prefix
338  self._db_type = db_type
339  self._metadata_handler = metadata_handler
340 
341  def _task_group(self, func, *args, **kw):
342  assert self._node_managers is not None, 'init must be called first.'
343  with TaskGroup(WorkspaceType.GLOBAL) as task_group:
344  for node, manager in self._node_managers:
345  with Node(node):
346  func(manager, *args, **kw)
347  return task_group
348 
349  """
350  Args:
351  nodes: An array of nodes where this checkpoint manager is running.
352  retrieve_from_epoch: Set to a number to load blobs from this epoch.
353  path_prefix: Used to construct db name or path where checkpoint files are
354  stored.
355  path_type: Indicate the type of path where checkpoint files are stored.
356  """
357  def init(
358  self, nodes, retrieve_from_epoch=None, path_prefix=None, path_type=None
359  ):
360  if self._node_managers is not None:
361  assert [node for node, _ in self._node_managers] == nodes
362  return
363  self._node_managers = []
364  self._path_prefix = path_prefix
365  self._path_type = path_type
366  self._node_names = [str(node) for node in nodes]
367  for node in nodes:
368  with Node(node):
369  manager = CheckpointManager(
370  db_prefix=self._db_prefix,
371  node_name=str(node),
372  db_type=self._db_type)
373  self._node_managers.append((node, manager))
374  return self._task_group(
375  CheckpointManager.init,
376  nodes=[node],
377  retrieve_from_epoch=retrieve_from_epoch,
378  path_prefix=path_prefix,
379  path_type=path_type)
380 
381  def load(self, epoch, path_prefix=None, path_type=None):
382  return self._task_group(
383  CheckpointManager.load,
384  epoch,
385  path_prefix=path_prefix,
386  path_type=path_type)

388  def load_blobs_locally(self, nodes, blob_names, epoch, session):
389  """Loads the necessary blobs from the checkpoints to the current node.
390 
391  Args:
392  blob_names: A list of strings. Each string is the name of a
393  blob.
394  epoch: An integer. The checkpoint epoch to load from.
395  session: A Session object to execute the Load ops.
396  """
397  if self._node_managers is not None:
398  assert [node for node, _ in self._node_managers] == nodes
399  else:
400  self._node_managers = []
401  for node in nodes:
402  with Node(node):
403  manager = CheckpointManager(
404  db_prefix=self._db_prefix,
405  node_name=str(node),
406  db_type=self._db_type)
407  self._node_managers.append((node, manager))
408  assert self._node_managers is not None, 'must initialize node managers'
409  for _, manager in self._node_managers:
410  existence_task = manager.check_db_exists(epoch)
411  session.run(existence_task)
412  existence = existence_task.outputs()[0].fetch()
413  if not existence:
414  logger.info('DB %s does not exist!' %
415  db_name(epoch, manager._node_name, manager._db_prefix))
416  return False
417  load_task = manager.load_blobs_from_checkpoint(blob_names, epoch)
418  session.run(load_task)
419  logger.info('Successfully loaded from checkpoints.')
420  return True

479 class JobRunner(object):
480  """
481  Implement the runtime logic for jobs with checkpointing at the level of
482  epoch. Can be used to run either single-host or distributed jobs. Job
483  runner is a callable to be called once from the master, passing a session
484  as an argument. This call will block until the Job execution is complete.
485 
486  If a checkpoint_manager is passed, checkpoints will be taken after
487  initialization and after each epoch execution. If, in addition,
488  `resume_from_epoch` is an epoch number, the corresponding checkpoint will
489  be loaded and job execution will continue from the given epoch. In
490  this case, the job's init_group will not be run.
491 
492  Refer to checkpoint_test.py for an example.
493  """
494  def __init__(self, job, checkpoint_manager=None, resume_from_epoch=None,
495  upload_task_group_builder=None):
496  """Initializes the JobRunner.
497 
498  Args:
499  job: A Job object. The job to be executed.
500  checkpoint_manager: Can be a CheckpointManager for single machine
501  or a MultiNodeCheckpointManager for multi-machine. The manager
502  that initializes/saves/loads checkpoints.
503  resume_from_epoch: An integer. The epoch to resume from.
504  upload_task_group_builder: A subclass of the
505  UploadTaskGroupBuilder. Creates a task group to upload
506  checkpoints.
507  """
508  self.resume_from_epoch = resume_from_epoch
509  self.checkpoint_manager = checkpoint_manager
510  self.job = job
511  self.upload_task_group_builder = upload_task_group_builder
512 
513  def __call__(self, session):
514  """Runs the training flow.
515 
516  Args:
517  session: A Session object. Valid choises are: LocalSession,
518  LocalHostScheduler, and DistributedSession. It is used to
519  execute one TaskGroup a time.
520  """
521  # identify the epoch we must resume from
522  if self.checkpoint_manager:
523  self.resume_from_epoch = self.checkpoint_manager.\
524  get_resume_from_epoch_id(self.resume_from_epoch)
525  if self.resume_from_epoch is not None:
526  logger.info('Resuming from epoch {}'.format(self.resume_from_epoch))
527 
528  # Initialize all the nodes.
529  from_scratch = self.resume_from_epoch is None
530  if from_scratch:
531  session.run(self.job.init_group)
532 
533  if self.checkpoint_manager:
534  logger.info('Preparing checkpoints ...')
535  session.run(self.checkpoint_manager.init(
536  self.job.nodes_to_checkpoint(),
537  retrieve_from_epoch=self.resume_from_epoch))
538  # Save the first checkpoint before training starts, or resume from
539  # a previously saved checkpoint.
540  if from_scratch:
541  self.save_checkpoints(0, session)
542  else:
543  logger.info('Loading checkpoints for epoch {} ...'.format(
544  self.resume_from_epoch))
545  session.run(
546  self.checkpoint_manager.load(self.resume_from_epoch))
547  logger.info('Checkpoint loaded')
548 
549  logger.info("Finished initializing")
550 
551  # Start training.
552  epoch = 1 if from_scratch else self.resume_from_epoch + 1
553  while True:
554  logger.info('Starting epoch %d' % epoch)
555  session.run(self.job.epoch_group)
556  logger.info('Finished epoch %d' % epoch)
557  stop_signals = [o.fetch() for o in self.job.stop_signals]
558 
559  if self.checkpoint_manager:
560  self.save_checkpoints(epoch, session)
561 
562  if any(stop_signals):
563  logger.info('Stopping')
564  break
565  epoch += 1
566  logger.info('Finished training')
567  # Upload the checkpoints.
568  if (self.upload_task_group_builder):
569  upload_task_group = self.upload_task_group_builder.build(
570  epoch, self.checkpoint_manager)
571  session.run(upload_task_group)
572  logger.info('Finished uploading the checkpoints')
573  session.run(self.job.exit_group)
574  logger.info('Finished running the exit group')
575  return epoch

577  def load_blobs_from_checkpoints(self, blob_names, epoch, session):
578  """Loads the necessary blobs from the checkpoints.
579 
580  Checkpoints store the snapshots of the workspace in each node.
581  Sometimes we only need to load a subset of the blobs from the
582  checkpoints. One common scenario is to load only the model blobs from
583  the checkpoints for evaluation purpose. Given the names of the
584  necessary blobs, this function goes over all the checkpoints of all the
585  nodes, but only loads the blobs specified in the blob_names to the
586  current workspace.
587 
588  Args:
589  blob_names: A list of strings. Each string is the name of a
590  blob.
591  epoch: An integer. The checkpoint epoch to load from.
592  session: A Session object to execute the load ops.
593 
594  Raises:
595  ValueError: When the checkpoint manager is invalid.
596  """
597  if not self.checkpoint_manager:
598  raise ValueError('Checkpoint manager is None')
599  logger.info('Loading checkpoint for epoch {} ...'.format(epoch))
600  return self.checkpoint_manager.load_blobs_locally(
601  self.job.nodes_to_checkpoint(), blob_names, epoch, session)

606  This method will trigger the Save ops to serialize and persist the
607  blobs present in the global workspaace.
608 
609  Args:
610  epoch: An integer. The checkpoint epoch-id that we are saving.
611  session: A Session object to execute the save ops.
612 
613  Raises:

615  """

617  raise ValueError('Checkpoint manager is None')

628 def epoch_limiter(num_epochs):
629  """
630  Creates a task that will output True when a given
631  number of epochs has finished.
632  """
633  with Job.current().init_group:
634  init_net = core.Net('epoch_counter_init')
635  counter = init_net.CreateCounter([], init_count=num_epochs - 1)
636  Task(step=init_net)
637  epoch_net = core.Net('epoch_countdown')
638  finished = epoch_net.CountDown(counter)
639  output = Task(step=epoch_net, outputs=finished).outputs()[0]
640  Job.current().add_stop_signal(output)
