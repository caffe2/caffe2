134  def _gather_wrapper(self, net, version, in_indices, out):
135  # Gather can work on all kinds of input data types, and output
136  # data with the same type. Convert the output of Gather to float,
137  # because the follow-up Ops expect fp32.
138  if version == 'fp32':
139  return net.Gather([self.w, in_indices], out)
140  elif version == 'fp16':
141  gathered_w = net.Gather([self.w, in_indices], 'gathered_w')
142 
143  return net.HalfToFloat(gathered_w, out)
144  elif version == 'uint8rowwise':
145  gathered_w = net.Gather([self.w, in_indices], 'gathered_w')

148  'gathered_scale_bias'
149  )

152  [gathered_w, gathered_scale_bias], out)
