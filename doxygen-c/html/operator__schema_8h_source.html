400 class OpSchemaRegistry {

402  static OpSchema&
403  NewSchema(const string& key, const string& file, const int line) {
404  auto& m = map();
405  if (m.count(key)) {
406  const auto& schema = m[key];
407  std::ios_base::Init init;
408  std::cerr << "Trying to register schema with name " << key
409  << " from file " << file << " line " << line
410  << ", but it is already registered from file " << schema.file()
411  << " line " << schema.line();
412  abort();
413  }
414  m.emplace(std::make_pair(key, OpSchema(file, line)));
415  return m[key];
416  }

477 OpSchema::Cost PointwiseCostInference(
478  const OperatorDef& /* unused */,
479  const vector<TensorShape>& inputs) {
480  struct OpSchema::Cost c;
481  const TensorShape X = inputs[0];
482  uint64_t size = 1;
483 
484  for (auto i = 0; i < X.dims().size(); ++i) {
485  size *= X.dims(i);

488  c.flops = size * OpsPerPoint;
