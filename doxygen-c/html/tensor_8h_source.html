218  template <class ContextForCopy>
219  void Extend(TIndex num, float growthPct, ContextForCopy* context) {
220  CAFFE_ENFORCE_GE_WITH_CALLER(dims_.size(), 1);
221  auto newDims = dims_;
222  newDims[0] += num;
223  if (!data_) {
224  Resize(newDims);
225  return;
226  }
227  auto newSize = std::accumulate(
228  newDims.begin(),
229  newDims.end(),
230  static_cast<TIndex>(1),
231  std::multiplies<TIndex>());
232  if (newSize * meta_.itemsize() <= capacity_) {
233  dims_ = newDims;
234  size_ = newSize;
235  return;
236  }
237  auto newCapacity = dims_;
238  newCapacity[0] = std::max<size_t>(
239  newDims[0], std::ceil(dims_[0] * (growthPct + 100) / 100));
240  Reserve(newCapacity, context);
241  dims_ = newDims;
242  size_ = newSize;
243  }

245  template <class T, class ContextForCopy>
246  void Reserve(const std::vector<T>& newCapacity, ContextForCopy* context) {
247  auto newSize = std::accumulate(
248  newCapacity.begin(),
249  newCapacity.end(),
250  static_cast<TIndex>(1),
251  std::multiplies<TIndex>());
252  if (newSize * meta_.itemsize() <= capacity_) {
253  return;
254  }
255  auto oldData = std::move(data_);
256  auto oldSize = size_;
257  auto oldDims = dims_;
258  Resize(newCapacity);
259  auto* newData = raw_mutable_data(meta_);
260  context->template CopyItems<ContextForCopy, ContextForCopy>(
261  meta_, oldSize, oldData.get(), newData);
262  dims_ = oldDims;
263  size_ = oldSize;
264  reserved_ = true;
265  }

299  template <typename... Ts>
300  void Resize(Ts... dim_source) {
301  bool size_changed = SetDims(dim_source...);
302  if (size_changed) {
303  // If needed, we will free the data. the next mutable_data() call
304  // will create the data storage.
305  int64_t new_size = size_ * meta_.itemsize();
306  bool reset_tensor = false;
307  if (reserved_) {
308  // If tensor is reserved then don't claim its memeory unless capacity_
309  // is smaller than new size
310  reset_tensor = capacity_ < new_size;
311  } else {
312  reset_tensor = capacity_ < new_size || !FLAGS_caffe2_keep_on_shrink ||
313  capacity_ - new_size > FLAGS_caffe2_max_keep_on_shrink_memory;
314  }
315 
316  if (reset_tensor) {
317  FreeMemory();
318  }
319  }
320  }

444  template <typename Deleter = MemoryDeleter>
445  void ShareExternalPointer(
446  void* src,
447  const TypeMeta& meta,
448  size_t capacity = 0,
449  Deleter d = nullptr) {
450  meta_ = meta;
451  CAFFE_ENFORCE_WITH_CALLER(
452  meta_.id(),
453  "To share with a raw external pointer you need to have meta "
454  "already set.");
455  CAFFE_ENFORCE_WITH_CALLER(
456  size_ >= 0,
457  "To share data with a raw pointer, you need to set shape first.");
458  // Check if the deleter is a MemoryDeleter and is a simple nullptr.
459  if (std::is_same<MemoryDeleter, Deleter>::value &&
460  reinterpret_cast<MemoryDeleter*>(&d)[0] == nullptr) {
461  // Use aliasing constructor trick to avoid calling the destructor.
462  data_ = std::shared_ptr<void>(std::shared_ptr<void>(), src);
463  } else {
464  data_.reset(src, d);
465  }
466  // Sets capacity. If not specified, we will implicitly assume that
467  // the capacity is the current size.
468  if (capacity) {
469  capacity_ = capacity;
470  } else {
471  capacity_ = nbytes();
472  }
473  shares_data_ = true;
474  }

522  inline void* raw_mutable_data(const TypeMeta& meta) {
523  // For 0-size tensors it's fine to return any pointer (including nullptr)
524  if (meta_ == meta && (data_.get() || size_ == 0)) {
525  return data_.get();
526  } else {
527  bool had_special_dtor = meta_.dtor() != nullptr;
528  meta_ = meta;
529  CAFFE_ENFORCE_WITH_CALLER(
530  size_ >= 0,
531  "Tensor is not initialized. You probably need to call Resize() "
532  "before calling mutable_data()");
533 
534  // We can reuse the existing buffer if the current data does not have
535  // a special destructor and the new data doesn't have a special
536  // constructor.
537  if (size_ == 0 ||
538  (meta.ctor() == nullptr && !had_special_dtor &&
539  capacity_ >= size_ * meta_.itemsize())) {
540  return data_.get();
541  }
542  if (meta.ctor()) {
543  // For types that need placement new, we will call it, as well as
544  // making sure that when the data is freed, it calls the right
545  // destruction procedure.
546  auto size = size_;
547  auto dtor = meta_.dtor();
548  auto ptr_and_deleter = Context::New(size_ * meta_.itemsize());
549  auto deleter = ptr_and_deleter.second;
550  data_.reset(
551  ptr_and_deleter.first, [size, dtor, deleter](void* ptr) -> void {
552  dtor(ptr, size);
553  deleter(ptr);
554  });
555  meta_.ctor()(data_.get(), size_);
556  } else {
557  // For fundamental type, new and delete is easier.
558  auto ptr_and_deleter = Context::New(size_ * meta_.itemsize());
559  data_.reset(ptr_and_deleter.first, ptr_and_deleter.second);
560  }
561  capacity_ = size_ * meta_.itemsize();
562  return data_.get();
563  }
564  }

833 template <class T>
834 void TensorPrinter::Print(const Tensor<CPUContext>& tensor) {
835  std::stringstream values_stream;
836  // One most likely doesn't want to print int64-number of items for visual
837  // inspection, so we cast down to int here.
838  int total_count = static_cast<int>(
839  std::min(tensor.size(), TIndex(limit_)));
840  const T* tensor_data = tensor.template data<T>();
841  for (int i = 0; i < total_count - 1; ++i) {
842  values_stream << tensor_data[i] << ",";
843  }
844  // We do not add a comma after the last item.
845  values_stream << tensor_data[total_count - 1];
846  if (to_file_) {
847  (*log_file_) << MetaStr(tensor) << values_stream.str() << std::endl;
848  } else {
849  // Log to console.
850  LOG(INFO) << MetaStr(tensor) << values_stream.str();
851  }
